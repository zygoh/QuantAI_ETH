```python
# ml_service.py
class EnsembleMLService:
    def __init__(self):
        self.lgb_models = {}
        self.xgb_models = {}
        self.cat_models = {}
    
    async def predict_ensemble(self, X, timeframe):
        # 三模型投票
        lgb_prob = self.lgb_models[timeframe].predict_proba(X)
        xgb_prob = self.xgb_models[timeframe].predict_proba(X)
        cat_prob = self.cat_models[timeframe].predict_proba(X)
        
        # 加权平均
        ensemble = lgb_prob * 0.4 + xgb_prob * 0.3 + cat_prob * 0.3
        return ensemble
```
